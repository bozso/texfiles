\newcommand{\Name}[2] {%
    \begin{tikzpicture}
        \draw node[rectangle, minimum height=3.6cm,
                   minimum width=7cm, rounded corners=0pt] at (0,0) {%
            \begin{minipage}{6cm}
            \begin{center} {
                \bfseries
                %\vspace{-20pt}
                { \huge  #1 }
                \vspace{10pt}
                
                { \large #2 }
            }
            \end{center}
            \end{minipage}
        };
    \end{tikzpicture}
}
