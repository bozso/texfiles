\newcommand{\Name}[3] {%
    \begin{tikzpicture}
        \draw node[rectangle, shade, draw=\mcol, top color=white,
                   bottom color=\mcol, shading=axis, shading angle=#3,
                   minimum height=3.6cm, minimum width=7cm] at (0,0) {%
            \begin{minipage}{6cm}
            \begin{center} {
                \bfseries
                %\vspace{-20pt}
                
                { \huge \color{blue!25!black} #1 }
                \vspace{15pt}
                
                { \large \color{white!95!blue} #2 }
            }
            \end{center}
            \end{minipage}
        };
    \end{tikzpicture}
}
