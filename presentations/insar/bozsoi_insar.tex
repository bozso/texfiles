    Goldstein-szűrés:
    \[
        H(u,v) = Z(u,v) \abs{Z(u,v)}^{\alpha}
    \]
    \begin{itemize}
        \item $H(u,v)$: szűrt értékek
        \item $Z(u,v)$: eredeti értékek
        \item $\alpha$: szűrés erőssége; $\alpha = 0$ nincs szűrés
    \end{itemize}
\end{frame}
